Integration Tutorial (Roman Urdu)

API Kaise Use Karein?

  1. Sabse pehle upar diye gaye button se apni API Key generate karein.
  2. Apne code mein humare endpoint https://api.qreator.studio/v1/generate par GET request bhejein.
  3. Parameters mein apikey, text, color, aur logo pass karein.
  4. API response mein direct high-resolution PNG Image return karegi jise aap direct display kar sakte hain.
// 1. HTML Image Element
const imgElement = document.getElementById('my-qr-code');

// 2. API Parameters
const apiKey = 'YOUR_API_KEY';
const text = encodeURIComponent('https://unsplash.com');
const color = '0f172a';
const logo = 'instagram';

// 3. Set Image Source directly to API URL
imgElement.src = `https://api.qreator.studio/v1/generate?apikey=${apiKey}&text=${text}&color=${color}&logo=${logo}`;

import requests

api_key = "YOUR_API_KEY"
url = "https://api.qreator.studio/v1/generate"

params = {
    "apikey": api_key,
    "text": "https://unsplash.com",
    "color": "0f172a",
    "logo": "instagram"
}

response = requests.get(url, params=params)

if response.status_code == 200:
    with open("my_qr.png", "wb") as f:
        f.write(response.content)
    print("QR Code saved successfully!")

// Flutter Image Widget with Network URL
Image.network(
  'https://api.qreator.studio/v1/generate' +
  '?apikey=YOUR_API_KEY' +
  '&text=https://unsplash.com' +
  '&color=0f172a' +
  '&logo=instagram',
  loadingBuilder: (context, child, loadingProgress) {
    if (loadingProgress == null) return child;
    return CircularProgressIndicator();
  },
  errorBuilder: (context, error, stackTrace) {
    return Text('Failed to load QR Code');
  },
);
